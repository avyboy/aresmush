def afk_message
  if (@char.is_afk)
    format_afk_message
  elsif (Status.is_idle?(@client))
    "%xy%xh<#{t('describe.idle')}>%xn"
  else
    ""
  end
end

def format_afk_message
  msg = "%xy%xh<#{t('describe.afk')}>%xn"
  if (@char.afk_message)
    msg = "#{msg} %xy#{@char.afk_message}%xn"
  end
  msg
end
